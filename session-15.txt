#!/bin/bash

START_TIME=$(date +%s)

echo "Script executed at: $START_TIME"

sleep 10

TOTAL_TIME=$(($END_TIME-$START_TIME))

echo "Script executed in: $TOTAL_TIME seconds"

09-conditions.sh
----------------
#!/bin/bash

NUMBER=$1  #we have to check whether the given number is greater than 20

# -gt -> greater than
# -lt -> less than
# -eq -> equal
# -ne -> not equal
if [ $NUMBER -gt 20 ]; then
    echo "Given number: $NUMBER is greater than 20" #if above expression true this will be executed and exits from loop
elif [ $NUMBER -eq 20 ]; then
    echo "Given number: $NUMBER is equal to 20"  #if above expression false this will be executed and exits from loop
else
    echo "Given number: $NUMBER is less than 20"  #if above both are false this will be executed and exits from loop
fi

output
