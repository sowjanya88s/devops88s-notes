#!/bin/bash

USERID=$(id -u)
LOGS_FOLDER="/var/log/shell-roboshop"
LOGS_FILE="$LOGS_FOLDER/$0.log"
R="\e[31m"
G="\e[32m"
Y="\e[33m"
N="\e[0m"

if [ $USERID -ne 0 ]; then
    echo -e "$R Please run this script with root user access $N" | tee -a $LOGS_FILE
    exit 1
fi

mkdir -p $LOGS_FOLDER

VALIDATE(){
    if [ $1 -ne 0 ]; then
        echo -e "$2 ... $R FAILURE $N" | tee -a $LOGS_FILE
        exit 1
    else
        echo -e "$2 ... $G SUCCESS $N" | tee -a $LOGS_FILE
    fi
}

dnf module disable redis -y &>>$LOGS_FILE
dnf module enable redis:7 -y &>>$LOGS_FILE
VALIDATE $? "Enable Redis:7"

dnf install redis -y  &>>$LOGS_FILE
VALIDATE $? "Installed Redis"

sed -i -e 's/127.0.0.1/0.0.0.0/g' -e '/protected-mode/ c protected-mode no' /etc/redis/redis.conf
VALIDATE $? "Allowing remote connections"

systemctl enable redis &>>$LOGS_FILE
systemctl start redis 
VALIDATE $? "Enabled and started Redis"

How to call one script from another script
$$ -- gets current process instance id
to call script-2 from script-1 we have two methods
1. calling directly using sh command -- sh <script-name>.sh
2. calling by using source command 


sh script-2.sh -> seperate process, can't access script-1 variables -- gives different process ids
source ./22-script-2.sh -> same process, can access script-1 variables -- gives same process ids

when u give any command -- linux checks that command is available in any of below paths

give below command to know command related paths -- echo $PATH
/home/ec2-user/.local/bin:
/home/ec2-user/bin:
/usr/local/bin:
/usr/bin:
/usr/local/sbin:
/usr/sbin
